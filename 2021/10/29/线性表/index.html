0.0 初识

为了表示每个数据元素与其直接后继数据元素之间的逻辑关系，每个数据元素除了自身的信息以外，还需要存储一个指示其直接后继的信息（后继节点的存储位置）。这两部分信息组成数据元素的存储映像——节点。

其中存储元素数据信息的域: 数据域；存储直接后继 存储位置 的域: 指针域。n个节点链结成一个链表，为线性表的链式存储结构。

0.1操作习惯

  • 整个链表的存取必须从头指针开始进行。头指针: 链表中第一个节点的存储位置。

  • 单链表的头元素为单独的一个元素。其数据信息可以什么都不存储，或 保存整个链表的部分信息，

    ​ 如链表长度；或节点个数。

  • 要取得第i 个数据元素，必须从头指针出发顺链进行寻找。

    ​ 因单链表中，任何两个元素的存储位置之间没有固定的联系。每个元素的存储位置都保存在其直接前驱节点的信息里。

1. 顺序表 (C)

声明一个结构体。结构体内包含一个数组与一个int 型记录数组长度。然后自动初始化，再根据提示进行查找，删除，插入操作。

1.1 声明区

#include <stdio.h>
#include <stdlib.h>
#include <malloc.h>

#define maxsize 20//字符串的最大长度
#define OK 1    //操作OK时返回OK
#define error -1//操作不OK时返回error

typedef struct Lnode {
    int *elem;
    int length;
}Lnode;//定义结构体 Lnode

void list_init(Lnode* l);//数组初始化,给数组的每一项赋值。
void list_print(Lnode*l);//把整个数组打印出来,打印格式为: elem [index]: value。
int list_find(Lnode* l, int value);//根据 value ，返回这个值的序列或 -1 。
int list_insert(Lnode* l,int value,int index );//在数组末尾插入 value，超出最大范围返回-1。
int list_delete(Lnode *l, int index); //删除传入序列对应的值。//函数预声明。避免顺序问题

1.2 主函数

int main() {

    Lnode *ln = (Lnode*)malloc(sizeof(Lnode));
    list_init(ln);
    //初始化
    list_print(ln);
    printf("\n请输入要查找的值,将会输出他的序列:\n");
    int value;
    scanf("%d", &value);
    //查找
    int term=list_find(ln, value) ;
    if (term< 0)
        printf("查找的值不存在");
    else {
        printf("所查找的值的序列为%d\n", term);
    }
    //删除
    list_print(ln);
    int index;
    printf("\n请输入删除的序列:\n");
    scanf("%d", &index);
    printf(((list_delete(ln, index)==OK)?"删除成功":"删除失败"));
    printf("\n删除后数组:");
    list_print(ln);
    //插入
    printf("\n输入将要插入的值与序列，之间以空格隔开:\n");
    value = NULL;
    index = NULL;
    scanf("%d %d", &value,&index);
    printf(((list_insert(ln,value,index-1)==OK)?"插入成功":"插入失败"));
    printf("\n插入后数组为:\n");
    list_print(ln);
}

1.3 初始化函数

void list_init(Lnode* l) {
    //根据传入的指针，初始化此结构体内数组；
    l->elem = (int*)malloc(sizeof(int[maxsize]));

    if (!l->elem) exit(0);
    l->length = 0;

    for (int i = 0; i < maxsize; i++)
    {
        (l->elem)[i] = i * 2;
        l->length++;
    }
}

1.4输出函数

void list_print(Lnode*l) {
    //格式:每五个换行
    //       elem[0  ]:0
    for (int i = 0; i < l->length; i++)
    {
        if (i % 5 == 0)printf("\n");
        printf("elem[%-3d]:%-3d  ",i, l->elem[i]);
    }
}

1.5 查找

int list_find(Lnode* l,int value) {
    //传入值value，输出序列至x；
    for (int i = 0; i < l->length; i++)
    {
        if (l->elem[i] == value) {
            return i;
        }
    }
    return error;
}

1.6 插入

int list_insert(Lnode* l,int value,int index ) {
    //刚满,或不足,或序列不合规返回error。
    if (l->length == maxsize||index>maxsize||index<0) {
        return error;
    }
    else {
        l->length++;
        for (int i = l->length;  i > index;  i--)
        {
            l->elem[i] = l->elem[i - 1];
            l->elem[index] = value;
        }
        l->elem[l->length-1] = value;
        return OK;
    }
}

1.7 删除

int list_delete(Lnode *l, int index) {
    if (index<0 || index>maxsize - 2) {
        return error;
    }
    else
    {
        for (int j = index; j < l->length - 1; j++)
        {
            l->elem[j] = l->elem[j + 1];
        }
        l->length--;
        return OK;
    }
}

typedef struct TEST {

int init(TEST * head);

2.2 main函数


int main() {
    TEST * head = NULL;
    //链表头
    int n=0;
    int pos=1;

    if (init(&head) == 0) {
        printf("创建失败");
        return 0;
    }
    else{
        printf("请输入链表节点数:\n");
        scanf_s("%d", &n);
        if (create(&head, n) == 0) {
            printf("添加数据失败");
            return 0;
        }
    }
    printNode(head);
    //输出
    printf("\n\n再添加一个\n");
    //插入
    printf("请输入插入位置（0~%d之间）:\n",head->num+2);    
    scanf_s("%d", &pos);
    if (pos<1 || pos>head->num + 2) {
        printf("输入有误,将跳过插入操作。");
    }
    else {
        int value = 0;
        printf("输入插入的值(别输太大)：\n");
        scanf_s("%d", &value);
        insertEle(&head, pos, value);
    }
    printNode(head);
    //输出
    pos = 0;
    //删除
    printf("输入要删除的索引号(1~%d):\n", head->num);
    scanf_s("%d", &pos);
    if (pos<1 || pos>head->num) {
        printf("输的什么鬼！，直接跳过删除。");
    }
    else {
        deleteEle(&head, pos);
        //删除一个元素
        printf("\n删除一个数据后:\n");
    }
    printNode(head);
    //输出
}

2.3 初始化

int init(TEST * head) {
    //加个链表头。
    head->next = NULL;
    head->num = 0;
    if (head == NULL) {
        return 0;
    }
    else {
        return 1;
    }
}

2.4 创建

int create(TEST * head, int n) {
    //链表添加数据，共 n 个
    TEST *q;
    TEST* p;
    head->num = n;
    q = head;

    for (int i = 1; i <= n; i++) {
        p = (struct TEST*)malloc(sizeof(TEST));
        p->num = i;
        p->next = NULL;
        q->next = p;
        q = p;
    }
    if ((head)->num != n) {
        return 0;
    }
    else {
        return 1;
    }
}

2.5 打印输出

void printNode(TEST* head) {
    //遍历输出打印.

    if (head->num == 0) {
        printf("\n\n一个空链表。没啥输出的。\n");
        return;
    }
    else {
        int j = 0;
        for (TEST* i = head->next; j < head->num; i = i->next) {
            if (i != NULL) {
                printf("%d ", i->num);
                j++;
            }
            else {
                break;
            }
        }
        printf("\n");
    }
}

2.6 插入

int insertEle(TEST* head, int pos, int value) {
    TEST* q = head;
    TEST* p = q->next;

    if (pos <= ((head)->num + 1) && pos > 0) {
        pos--;
        while (pos != 0 && p != NULL) {
            q = p;
            p = q->next;
            pos--;
        }
        //找位置            
        q->next = (TEST*)malloc(sizeof(struct TEST));
        q->next->next = p;
        q->next->num = value;
        (head)->num++;
        //插数据。
        return 1;
    }
    else {
        return 0;
    }
}

2.7 删除

int deleteEle(TEST *head, int pos) {
    //删除指定的项.
    TEST* p;
    TEST* q;
    q = head;
    p = q->next;

    if (pos == 1) {
        head->next = p->next;
        free(p);
        head->num--;
        return 1;
    }
    else if (pos > head->num) {
        printf("输入序列超出索引。");
        return 0;
    }
    else {
        do
        {
            q = p;
            p = q->next;
            pos--;
        } while (p != NULL && pos != 1);
        if (pos == 1)
        {
            q->next = p->next;
            head->num--;
            free(p);
            return 1;
        }
        else {
            printf("链表出现问题!");
            return 0;
        }
    }
}

2.8 销毁

int destroy(TEST* head) {
    //全部销毁
    TEST* p = (head)->next;
    TEST* q = NULL;

    if (p == NULL) {
        free(head);
        head = NULL;
        return 1;
    }
    else {
        do {
            q = p;
            p = q->next;
            free(q);
            (head)->num--;
        } while (p != NULL);
        return 1;
    }
    return 0;
}

2.9 链表连接

void connect(TEST* a, TEST* b) {
    // 把b接在a后边。
    TEST*  q,*p;
    p = a;
    while(p->next)
    {
        p = p->next;
    }
    p->next = b->next;
    a->num += b->num;
}

3. 双向链表 (Java)

任何操作都需要考虑节点的先驱、后驱。

3.1 链表结构

/**
    * @param last 上个节点
    * @param next 下个链表
    * @param value 节点值
    */
public static class BothDirLinkList {
    BothDirLinkList last;
    BothDirLinkList next;
    String value;
    //无参构造
    public BothDirLinkList() {
    }
    //全参构造
    public BothDirLinkList(BothDirLinkList last, BothDirLinkList next, String value) {
        this.last = last;
        this.next = next;
        this.value = value;
    }
}

3.2 创建

/**
     * @param n 创建链表节点数
     * @return 返回第一个节点。
     */
static BothDirLinkList createLinkList(int n) {
    BothDirLinkList head = new BothDirLinkList();
    BothDirLinkList pTerm;
    head.value = "我是头";
    BothDirLinkList qTerm = head;
    for (int i = 1; i < n; i++) {
        pTerm = new BothDirLinkList();
        pTerm.value = Integer.toString(i);
        pTerm.last = qTerm;
        qTerm.next = pTerm;
        qTerm = pTerm;
    }
    qTerm.next = head;
    head.last = qTerm;
    return head;
}

3.4 插入

/**
     * @param lis 插入数据的链表
     * @param str 插入的值
     * @param pos 插入位置 可为负，从 head 开始寻找，负数为逆向寻找。
     */
static void insertLinkList(BothDirLinkList lis, String str, int pos) {
    BothDirLinkList term = lis;
    if (pos == 0) {
        printString("序列有问题！");
        //lis.next.last = new BothDirLinkList(lis, lis.next, str);
        //lis.next = lis.next.last;
    } else {
        int v = pos / Math.abs(pos);
        // 获取定位节点的移动方向。
        do {
            //头指针为 0 序列。所以从lis.next 开始计数。
            term = (pos > 0) ? term.next : term.last;
            pos -= v;
        } while (pos != 0);
        BothDirLinkList tNode = (v > 0) ? new BothDirLinkList(term.last, term, str) : new BothDirLinkList(term, term.next, str);
        /* 大于零：插入节点在定位节点之后。
             * 小于零：插入节点在定位节点之前。
             * */
        if (v > 0) {
            /*正向插入
                 *
                 * 定位节点 后节点 的前驱 为 插入节点
                 * 定位节点 的 后驱      为 插入节点
                 *   */
            term.last.next = tNode;
            term.last = tNode;
        } else {
            /* 反向插入
                 * 原理同上
                 * */
            term.next.last = tNode;
            term.next = tNode;
        }
    }
}

3.5 删除

/**
     * @param lis   操作数组
     * @param value 删除节点的值
     * @return 是否删除成功
     */
static boolean deleteLinkList(BothDirLinkList lis, String value) {
    BothDirLinkList term = lis;
    do {
        if (term.value.equals(value)) {
            //中间截断，返回。
            term.next.last = term.last;
            term.last.next = term.next;
            return true;
        } else {
            term = term.next;
        }
    } while (term != lis);
    return false;
}
/**
     * @param lis   操作数组
     * @param index 删除节点序列 非零整数
     * @return 是否删除成功
     */
static boolean deleteLinkList(BothDirLinkList lis, int index) {
    BothDirLinkList term = lis;
    if (index == 0) {
        return false;
    } else {
        int v = index / Math.abs(index);
        //获取定位节点的移动方向。
        do {
            term = (index > 0) ? term.next : term.last;
            index -= v;
        } while (index != 0);

        term.next.last = term.last;
        term.last.next = term.next;
        return true;
    }
}

/**
     * @param lis       链表第一个元素
     * @param isReverse 是否为逆序
     * @param times     循环打印次数
     */
static void printLinkList(BothDirLinkList lis, int times, boolean isReverse) {
    BothDirLinkList tr = lis;

    while (times >= 0) {
        if (tr == lis) {
            if (times == 0) {
                break;
            }
            times--;
        }
        System.out.println("value = " + tr.value);
        tr = isReverse ? tr.last : tr.next;
    }
    System.out.print("\n");
}
